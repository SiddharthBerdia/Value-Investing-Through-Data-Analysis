    function loginUser(event) {
        event.preventDefault();
        const email = document.getElementById('email').value;
        const password = document.getElementById('password').value;

        if(email && password) {
            document.getElementById('login').style.display = 'none';
            document.getElementById('home').style.display = 'none';
            document.getElementById('dashboard').style.display = 'block';
            document.getElementById('userEmail').innerText = email;
            loadSampleData();
        } else {
            alert('Please fill all fields');
        }
    }

    function loadSampleData() {
        const sampleStocks = [
            { stock: 'ABC Ltd', rv_score: '0.12', price: '$120' },
            { stock: 'XYZ Corp', rv_score: '0.15', price: '$98' },
            { stock: 'MNO Industries', rv_score: '0.18', price: '$230' }
        ];

        const tbody = document.getElementById('stockTableBody');
        tbody.innerHTML = '';

        sampleStocks.forEach(s => {
            const row = `<tr><td>${s.stock}</td><td>${s.rv_score}</td><td>${s.price}</td></tr>`;
            tbody.innerHTML += row;
        });
    }
